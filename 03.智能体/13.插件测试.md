<script src="//ptlib.tb-n.com/lib/sys/require.min.js"></script>
<script type="text/javascript">
/*
 * 数据管理模板
 */

// 数据配置
var objdata = {
    statusList: [
        {value: 0, text: "禁用", id: 0},
        {value: 1, text: "启用", id: 1},
        {value: 2, text: "待审核", id: 2}
    ]
};

// RequireJS配置
require.config({
    paths: {
        jquery: '../../sys/jquery',
        system: '../../sys/system',
        operationlog: '../../js/system/operationlog',
        layui: "../../layui-btkj/layui"
    },
    shim: {
        system: {
            deps: ["jquery"]
        }
    },
    waitSeconds: 0
});

// 主程序入口
require(["jquery", "system", "operationlog", "layui"], function () {
    layui.use(['table', 'form', 'laydate', 'layer'], function () {
        let table = layui.table;
        let form = layui.form;
        let layer = layui.layer;
        let laydate = layui.laydate;

        // 初始化状态选择器
        initStatusSelect(form);

        // 初始化时间选择器
        initDatePickers(laydate, layer);

        // 初始化表格
        initDataTable(table, form, layer);

        // 搜索框回车事件
        $(document).on('keydown', function (event) {
            if (event.keyCode == 13) {
                search(table);
                return false;
            }
        });

        // 搜索按钮
        $("#btnSearch").click(function () {
            search(table);
        });

        // 重置按钮
        $("#btnReset").click(function () {
            $("#selStatus").val("");
            $("#keyword").val("");
            $("#startTime").val("");
            $("#endTime").val("");
            form.render('select');

            table.reload('laytable', {
                where: {swhere: $.msgwhere()}
            });
        });

        // 新增按钮
        $("#btnAdd").click(function () {
            layer.msg('新增功能待实现', {icon: 1});
        });
    });
});

/*初始化状态选择器*/
function initStatusSelect(form) {
    var selectHtml = '<option value="">请选择状态</option>';
    
    $.each(objdata.statusList, function(index, item) {
        selectHtml += '<option value="' + item.value + '">' + item.text + '</option>';
    });
    
    $("#selStatus").html(selectHtml);
    form.render('select');
}

/*初始化时间选择器*/
function initDatePickers(laydate, layer) {
    var startTimeValue = '';
    var endTimeValue = '';

    // 开始时间
    laydate.render({
        elem: '#startTime',
        type: 'datetime',
        format: 'yyyy-MM-dd HH:mm:ss',
        max: '2099-12-31 23:59:59',
        done: function(value, date) {
            startTimeValue = value;
            if (value) {
                laydate.render({
                    elem: '#endTime',
                    type: 'datetime',
                    format: 'yyyy-MM-dd HH:mm:ss',
                    min: value,
                    max: '2099-12-31 23:59:59',
                    done: function(value, date) {
                        endTimeValue = value;
                        if (startTimeValue && endTimeValue) {
                            validateTimeRange(startTimeValue, endTimeValue, layer);
                        }
                    }
                });
            }
        }
    });

    // 结束时间
    laydate.render({
        elem: '#endTime',
        type: 'datetime',
        format: 'yyyy-MM-dd HH:mm:ss',
        max: '2099-12-31 23:59:59',
        done: function(value, date) {
            endTimeValue = value;
            if (startTimeValue && endTimeValue) {
                validateTimeRange(startTimeValue, endTimeValue, layer);
            }
        }
    });
}

/*验证时间范围*/
function validateTimeRange(startTime, endTime, layer) {
    var startDate = new Date(startTime);
    var endDate = new Date(endTime);

    if (startDate >= endDate) {
        layer.msg('开始时间必须小于结束时间！', {icon: 2});
        $("#endTime").val("");
        return false;
    }
    return true;
}

/*执行搜索*/
function search(table) {
    var startTime = $("#startTime").val();
    var endTime = $("#endTime").val();

    if (startTime && endTime) {
        if (!validateTimeRange(startTime, endTime, layui.layer)) {
            return;
        }
    }

    table.reload('laytable', {
        page: {
            curr: 1
        },
        where: {
            swhere: $.msgwhere(getwhere())
        }
    });
}

/*拼接搜索条件*/
function getwhere() {
    var objwhere = {};

    var selStatus = $("#selStatus").val();
    var keyword = $("#keyword").val();
    var startTime = $("#startTime").val();
    var endTime = $("#endTime").val();

    if (selStatus) {
        objwhere.status = [selStatus];
    }
    if (keyword) {
        objwhere.keyword = [keyword];
    }
    if (startTime && endTime) {
        objwhere.startime = [startTime];
        objwhere.endtime = [endTime];
    } else if (startTime) {
        objwhere.startime = [startTime];
    } else if (endTime) {
        objwhere.endtime = [endTime];
    }

    return objwhere;
}

/*初始化数据表格*/
function initDataTable(table, form, layer) {
    table.render({
        elem: "#laytable",
        url: encodeURI($.smurl + "?" + $.getSmStr(["your_module.getList"])),
        where: {
            swhere: $.msgwhere(getwhere()),
            fields: 'id',
            types: 'desc'
        },
        cols: [[
            {type: 'numbers', width: 50, title: '序号'},
            {field: 'id', title: 'ID', width: 80, align: 'center'},
            {field: 'name', title: '名称', width: 200, align: 'center'},
            {field: 'status', title: '状态', width: 100, align: 'center', templet: function(d) {
                var statusText = '';
                $.each(objdata.statusList, function(index, item) {
                    if (item.value == d.status) {
                        statusText = item.text;
                        return false;
                    }
                });
                return statusText || d.status;
            }},
            {field: 'description', title: '描述', minWidth: 300, align: 'center'},
            {field: 'creatime', title: '创建时间', width: 180, align: 'center', templet: function(d) {
                if (d.creatime) {
                    var date = new Date(d.creatime);
                    return date.getFullYear() + '-' +
                        String(date.getMonth() + 1).padStart(2, '0') + '-' +
                        String(date.getDate()).padStart(2, '0') + ' ' +
                        String(date.getHours()).padStart(2, '0') + ':' +
                        String(date.getMinutes()).padStart(2, '0') + ':' +
                        String(date.getSeconds()).padStart(2, '0');
                }
                return '';
            }},
            {fixed: 'right', title: '操作', width: 150, align: 'center', toolbar: '#toolbarTpl'}
        ]],
        done: function (res, curr, count) {
            // 表格渲染完成后的回调
        },
        height: 'full-' + ($("#laytable").offset().top + 40),
        countNumberBool: true,
        page: true,
        limits: [10, 20, 30, 50, 100],
        limit: 10
    });

    // 监听工具条
    table.on('tool(laytable)', function(obj) {
        var data = obj.data;
        if(obj.event === 'edit') {
            layer.msg('编辑功能待实现: ' + data.id, {icon: 1});
        } else if(obj.event === 'del') {
            layer.confirm('确定删除该数据吗？', function(index) {
                layer.msg('删除功能待实现: ' + data.id, {icon: 1});
                layer.close(index);
            });
        }
    });
}
</script>
